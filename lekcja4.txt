#zadanie 3.1

#zadanie 3.2
select kreatura.nazwa, sum(ekwipunek.ilosc) from zasob, ekwipunek, kreatura where ekwipunek.idZasobu=zasob.idZasobu and kreatura.idkreatury=ekwipunek.idkreatury group by kreatura.nazwa;

#zadanie 3.3
select nazwa,sum(ilosc) from kreatura left join ekwipunek on ekwipunek.idkreatury=kreatura.idkreatury group by nazwa;

#zadanie 4.1
select kreatura.nazwa, zasob.nazwa, sum(ekwipunek.ilosc) from kreatura, zasob, ekwipunek where dataUr like "167%" and ekwipunek.idkreatury=kreatura.idkreatury and zasob.idzasobu = ekwipunek.idzasobu group by kreatura.nazwa, zasob.nazwa;

#zadanie 4.2
select kreatura.nazwa,zasob.rodzaj, ekwipunek.idkreatury from kreatura,ekwipunek, zasob where (ekwipunek.idkreatury=kreatura.idkreatury and zasob.idzasobu=ekwipunek.idzasobu and zasob.rodzaj="jedzenie") group by kreatura.nazwa,zasob.rodzaj, ekwipunek.idkreatury,kreatura.dataur order by dataUr Desc limit 5;
